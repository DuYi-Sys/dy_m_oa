描述
提供对文件所有属性的访问。
说明
下面的代码演示了如何获得 File 对象以及如何查看它的一个属性。 
function ShowFileInfo(filespec)
{
  var fso, f, s;
  fso = new ActiveXObject("Scripting.FileSystemObject");
  f = fso.GetFile(filespec);
  s = f.DateCreated;
  return(s);
}
